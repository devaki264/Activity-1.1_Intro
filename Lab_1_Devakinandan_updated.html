    'asia': 'red',      # Red for Asia
    'africa': 'aqua',   # Aqua for Africa
    'americas': 'lime', # Lime for Americas
    'europe': 'yellow'  # Yellow for Europe
}

# Filter the dataset for only the four specified regions
filtered_data = data[data['four_regions'].isin(color_mapping.keys())]

# Ensure that the 'energy' column is numeric and handle missing values
filtered_data['energy'] = pd.to_numeric(filtered_data['energy'], errors='coerce').fillna(0)

# Create the plot using plotnine
plot = (
    ggplot(filtered_data, aes(x='exports', y='imports', size='energy', fill='four_regions')) +
    annotate('text', x=110, y=200, label='1997', size=150, alpha=0.1, color='grey') +  # Center and enlarge "1997"
    geom_point(alpha=0.7, color="black", stroke=0.5) +  # Add light black outline and keep color fills
    scale_size_continuous(range=(2, 12)) +
    scale_x_continuous(breaks=range(0, 240, 20), labels=lambda l: ["{}".format(v) for v in l], limits=(0, 220)) +  # Remove percentage symbol
    scale_y_continuous(breaks=range(0, 450, 50), limits=(0, 400)) +  # Set Y-axis breaks and limits
    scale_fill_manual(values=color_mapping) +  # Apply the custom fill color mapping
    theme_minimal() +
    labs(x='Exports ', y='Imports') +
    guides(size=False, fill=False)  # Disable legends for size and color

from plotnine import ggplot, aes, geom_bar, scale_fill_manual, theme_minimal, labs, theme
import pandas as pd

# Load your dataset
data = pd.read_csv('q3data.csv')

# Filter the dataset to remove missing values in relevant columns
filtered_data_q3 = data.dropna(subset=['internet_users', 'gdp', 'income', 'four_regions'])

# Create a plot using plotnine with geom_bar
plot = (
    ggplot(filtered_data_q3, aes(x='four_regions', fill='four_regions')) +

    # Bar plot to show count of observations by region
    geom_bar() +
    scale_fill_manual(values=color_mapping) +  # Apply color mapping for regions
    theme_minimal() +  # Use minimal theme
    labs(x='World Regions', y='Count of Observations') +  # Label the axes
    theme(figure_size=(11, 5))  # Set figure size using theme
)

# Show the plot
plot.show()

Geom_bar(): Useful for aggregating data into categorical bins.

  • Why it’s good: If you want to summarize and compare grouped data, for instance, showing the average GDP for different world regions.

  • Why it’s bad: It loses individual point-level detail, so it’s not ideal for scatter-type relationships
